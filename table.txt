

CREATE TABLE sms.shops (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 shop_name text NOT NULL,
 address TEXT,
 cr_on timestamp without time zone DEFAULT now() NOT NULL,
 up_on timestamp without time zone DEFAULT now() NOT NULL

);

CREATE TABLE sms.counters (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 shop_id TEXT REFERENCES sms.shops(row_id) ON DELETE CASCADE,
 counter_name TEXT,
 location TEXT,
 cr_on timestamp without time zone DEFAULT now() NOT NULL,
 up_on timestamp without time zone DEFAULT now() NOT NULL

);


CREATE TABLE sms.users (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 name TEXT,
 email TEXT UNIQUE,
 phone TEXT,
 password TEXT,
 role TEXT,
counter_id TEXT ,
cr_on timestamp without time zone DEFAULT now() NOT NULL,
up_on timestamp without time zone DEFAULT now() NOT NULL

);

CREATE TABLE sms.suppliers (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 supplier_name TEXT,
 phone TEXT,
 email TEXT,
 address TEXT,
 cr_on timestamp without time zone DEFAULT now() NOT NULL,
    up_on timestamp without time zone DEFAULT now() NOT NULL

);


CREATE TABLE sms.departments (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 department_name VARCHAR(120) NOT NULL,
 description TEXT,
 cr_on timestamp without time zone DEFAULT now() NOT NULL,
 up_on timestamp without time zone DEFAULT now() NOT NULL

);

CREATE TABLE sms.categories (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 department_id TEXT REFERENCES sms.departments(row_id),
 category_name TEXT,
 cr_on timestamp without time zone DEFAULT now() NOT NULL,
 up_on timestamp without time zone DEFAULT now() NOT NULL

);

CREATE TABLE sms.sweets (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 category_id TEXT REFERENCES sms.categories(row_id),
 sweet_name TEXT,
 shelf_life_days TEXT,
 cr_on timestamp without time zone DEFAULT now() NOT NULL,
 up_on timestamp without time zone DEFAULT now() NOT NULL

);


CREATE TABLE sms.inventory (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 counter_id TEXT REFERENCES sms.counters(row_id),
 sweet_id TEXT REFERENCES sms.sweets(row_id),
 quantity TEXT DEFAULT 0,
 expiry_date DATE,
 cr_on timestamp without time zone DEFAULT now() NOT NULL,
    up_on timestamp without time zone DEFAULT now() NOT NULL,
 UNIQUE(counter_id,sweet_id)
);

CREATE TABLE sms.stock_transactions (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 counter_id TEXT REFERENCES sms.counters(row_id),
 sweet_id TEXT REFERENCES sms.sweets(row_id),
 transaction_type TEXT,
 quantity TEXT,
 reference_id TEXT,
 notes TEXT,
 cr_on timestamp without time zone DEFAULT now() NOT NULL,
    up_on timestamp without time zone DEFAULT now() NOT NULL
);

//TODO
CHECK (transaction_type IN ('IN','OUT','RETURN','ADJUSTMENT'));

CREATE TABLE sms.orders (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 counter_id TEXT REFERENCES sms.counters(row_id),
 supplier_id TEXT REFERENCES sms.suppliers(row_id),
 order_status TEXT DEFAULT 'PENDING',
 order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
 cr_on timestamp without time zone DEFAULT now() NOT NULL,
 up_on timestamp without time zone DEFAULT now() NOT NULL

);

CREATE TABLE sms.order_items (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 order_id TEXT REFERENCES sms.orders(row_id) ON DELETE CASCADE,
 sweet_id TEXT REFERENCES sms.sweets(row_id),
 quantity TEXT,
 cr_on timestamp without time zone DEFAULT now() NOT NULL,
 up_on timestamp without time zone DEFAULT now() NOT NULL

);

CREATE TABLE sms.chalans (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 order_id TEXT REFERENCES sms.orders(row_id),
 supplier_id TEXT REFERENCES sms.suppliers(row_id),
 dispatch_date TIMESTAMP,
 transport_details TEXT,
 cr_on timestamp without time zone DEFAULT now() NOT NULL,
 up_on timestamp without time zone DEFAULT now() NOT NULL

);

CREATE TABLE sms.returns (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 order_id TEXT REFERENCES sms.orders(row_id),
 sweet_id TEXT REFERENCES sms.sweets(row_id),
 quantity TEXT,
 reason TEXT,
cr_on timestamp without time zone DEFAULT now() NOT NULL,
    up_on timestamp without time zone DEFAULT now() NOT NULL

);

CREATE TABLE sms.expiry_logs (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 counter_id TEXT REFERENCES sms.counters(row_id),
 sweet_id TEXT REFERENCES sms.sweets(row_id),
 quantity TEXT,
 reason TEXT,
cr_on timestamp without time zone DEFAULT now() NOT NULL,
    up_on timestamp without time zone DEFAULT now() NOT NULL

);

CREATE TABLE sms.notifications (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 user_id TEXT REFERENCES sms.users(row_id),
 title VARCHAR(200),
 message TEXT,
 is_read BOOLEAN DEFAULT FALSE,
cr_on timestamp without time zone DEFAULT now() NOT NULL,
    up_on timestamp without time zone DEFAULT now() NOT NULL

);

CREATE TABLE sms.audit_logs (
 id SERIAL NOT NULL,
 row_id TEXT NOT NULL PRIMARY KEY,
 user_id TEXT REFERENCES sms.users(row_id),
 action VARCHAR(200),
 entity_type VARCHAR(50),
 entity_id TEXT,
 cr_on timestamp without time zone DEFAULT now() NOT NULL,
    up_on timestamp without time zone DEFAULT now() NOT NULL
);
